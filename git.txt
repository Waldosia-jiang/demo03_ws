//上传数据
git add .
git commit -m "commit"
git push -u origin main

//强制推送
git push --force


//生成秘钥
ssh-keygen -t rsa -b 4096 -C "your_email@example.com"
eval "$(ssh-agent -s)"
ssh-add ~/.ssh/id_rsa
ssh -T git@github.com

//将本地分支重置为远程分支的最新状态，这会覆盖本地的所有更改
git reset --hard origin/<branch-name>

//如何将本地仓库上传到远程仓库
git init
git add .
git commit -m "Initial commit"
git remote add origin <远程仓库的URL>
git push -u origin main

//将分支名的更改同步到远程仓库
git push origin -u <新分支名>

//修改仓库名称
git branch -m <新分支名>

//删除远程分支
git push origin --delete master

//在拉取更新时使用 rebase，这样会将你本地的提交应用在远程分支的最新提交之上
git pull --rebase origin main

//修改本地仓库名为main
git branch -m master main

//删除远程的 master 分支引用，并将 main 分支推送到远程
git push origin -u main
git push origin --delete master

//强制覆盖远程仓库
git push origin main --force


//删除已经上传并且被 Git 版本控制的 build 文件夹，你可以按照以下步骤操作
//将 build 文件夹添加到 .gitignore 文件中
build/

//从 Git 仓库中删除已经上传的 build 文件夹，但保留本地文件。运行以下命令
git rm -r --cached build/

//提交你的更改
git add .gitignore
git commit -m "Remove build folder from version control"

//推送到远程仓库
git push origin <branch-name>
